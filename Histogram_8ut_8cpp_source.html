33BOOST_AUTO_TEST_CASE(HistogramBasicCase)
34{
35 Histogram h{Lowest, Highest, Significant};

37 const auto expected_counts_len = Bitness == 64 ? 23552 : 22528;

39 BOOST_TEST(h.lowest_trackable_value() == Lowest);
40 BOOST_TEST(h.highest_trackable_value() == Highest);
41 BOOST_TEST(h.significant_figures() == Significant);
42 BOOST_TEST(h.sub_bucket_count() == 2048);
