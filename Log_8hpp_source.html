73 // Inspired by techniques developed by Rodrigo Fernandes.
74 class Log {
75  template <typename ValueT>
76  friend Log& operator<<(Log& log, ValueT&& val)
77  {
78  log.msg_ << std::forward<ValueT>(val);
79  return log;
80  }
81 
82  public:
83  enum : int {
85  Crit,
87  Error,
89  Warning,
91  Notice,
93  Info,
95  Debug
96  };
97 
98  explicit Log(int level) noexcept
99  : level_{level}
100  , msg_{log_msg()}
101  {
102  }
103  ~Log() { write_log(level_, msg_); }
104 
105  // Copy.
106  Log(const Log&) = delete;
107  Log& operator=(const Log&) = delete;
108 
109  // Move.
110  Log(Log&&) = delete;
111  Log& operator=(Log&&) = delete;
112 
113  constexpr explicit operator bool() const { return true; }
115  Log& operator()(const char* data, std::streamsize size)
116  {
117  msg_.write(data, size);
118  return *this;
119  }
121  Log& operator()() noexcept { return *this; }
122
