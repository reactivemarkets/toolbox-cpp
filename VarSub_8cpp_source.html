41 bool VarSub::substitute(string& s, const size_t i, size_t j, set<string>* outer) const
42 {
43  // Position of last substitution.
44  size_t last{0};
45  // Names substituted at 'last' position.
46  set<string> inner;
47 
48  int state{0};
49  while (j < s.size()) {
50  if (state == '\\') {
51  state = 0;
52  // Remove backslash.
53  s.erase(j - 1, 1);
54  } else {
55  const auto ch = s[j];
56  if (state == '$') {
57  state = 0;
58  if (ch == '{') {
59  if (j > last) {
60  // Position has advanced.
61  last = j;
62  inner.clear();
63  }
64  // Reverse to '$'.
65  --j;
66  // Descend: search for closing brace and substitute.
67  if (!substitute(s, j, j + 2, &inner)) {
68  return false;
69  }
70  continue;
71  }
72  }
73  switch (ch) {
74  case '$':
75  case '\\':
76  state = ch;
77  break;
78  case '}':
79  // If outer is null then the closing brace was found at the top level. I.e. there is no
80  // matching opening brace.
81  if (outer) {
82  // Substitute variable.
83  const auto n = j - i;
84  auto name = s.substr(i + 2, n - 2);
85  if (outer->count(name) == 0) {
86  s.replace(i, n + 1, fn_(name));
87  outer->insert(move(name));
88  } else {
89  // Loop detected: this name has already been substituted at this position.
90  s.erase(i, n + 1);
91  }
92  // Ascend: matched closing brace.
93  return true;
94  }
95  break;
96  }
97  }
98  ++j;
99  }
100  // Ascend: no closing brace.
101  return false;
102 }
