51 int64_t value_at_percentile(const HdrHistogram& h, double percentile) noexcept
52 {
53  const int64_t count_at_percentile{get_count_at_percentile(h, percentile)};
54 
55  int64_t total{0};
56  HdrIterator iter{h};
57  while (iter.next()) {
58  total += iter.count();

80 double stddev(const HdrHistogram& h) noexcept
81 {
82  const int64_t total_count{h.total_count()};
83  const double mean_val{mean(h)};
84 
85  double geometric_dev_total{0.0};
86  HdrIterator iter{h};
87  while (iter.next()) {
88  if (iter.count() != 0) {
89  const double dev{h.median_equivalent_value(iter.value()) - mean_val};
90  geometric_dev_total += (dev * dev) * iter.count();
91  }
92  }
93  return sqrt(geometric_dev_total / total_count);
94 }
