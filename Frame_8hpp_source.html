87 template <typename FnT>
88 std::size_t parse_frame(ConstBuffer buf, FnT fn)
89 {
90  std::size_t consumed{0};
91  for (;;) {
92  const auto* data = buffer_cast<const char*>(buf);
93  const std::size_t size = buffer_size(buf);
94  if (size < sizeof(std::uint32_t)) {
95  break;
96  }
97  const auto total = sizeof(std::uint32_t) + get_length(data);
98  if (size < total) {
99  break;
100  }
101  fn(ConstBuffer{data + sizeof(std::uint32_t), total - sizeof(std::uint32_t)});
102  buf = advance(buf, total);
103  consumed += total;
104  }
105  return consumed;
106 }
