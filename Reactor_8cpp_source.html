42Reactor::Handle Reactor::subscribe(int fd, unsigned events, IoSlot slot)
43{
44 assert(fd >= 0);
45 assert(slot);
46 if (fd >= static_cast<int>(data_.size())) {
47 data_.resize(fd + 1);
48 }
49 auto& ref = data_[fd];
50 epoll_.add(fd, ++ref.sid, events);
51 ref.events = events;
52 ref.slot = slot;

56int Reactor::poll(CyclTime now, Duration timeout)
57{
58 enum { High = 0, Low = 1 };
59 using namespace chrono;
60
61 // If timeout is zero then the wait_until time should also be zero to signify no wait.
62 MonoTime wait_until{};
63 if (!is_zero(timeout) && hooks_.empty()) {
64 const MonoTime next
65 = next_expiry(timeout == NoTimeout ? MonoClock::max() : now.mono_time() + timeout);
66 if (next > now.mono_time()) {
67 wait_until = next;
68 }
69 }
70 // TODO: consider using a dynamic buffer that scales with increased demand.
71 Event buf[MaxEvents];
72
73 int n;
74 error_code ec;
75 if (wait_until < MonoClock::max()) {
76 // The wait function will not block if time is zero.
77 n = epoll_.wait(buf, MaxEvents, wait_until, ec);
78 } else {
79 // Block indefinitely.
80 n = epoll_.wait(buf, MaxEvents, ec);
81 }
82 // Update cycle time after epoll() returns.
83 now = CyclTime::now();
84 if (ec) {
85 if (ec.value() != EINTR) {
86 throw system_error{ec};
87 }
88 return 0;
89 }
90 int work{0};
91 TOOLBOX_PROBE_SCOPED(reactor, dispatch, work);
92 // High priority timers.
93 work = tqs_[High].dispatch(now);
94 // I/O events.
95 work += dispatch(now, buf, n);
96 // Low priority timers are only dispatched during empty cycles.
97 if (work == 0) {
98 work += tqs_[Low].dispatch(now);
99 }
100 // End of cycle hooks.
101 io::dispatch(now, hooks_);
102 return work;
103}

112MonoTime Reactor::next_expiry(MonoTime next) const
113{
114 enum { High = 0, Low = 1 };
115 using namespace chrono;
116 {
117 const auto& tq = tqs_[High];
118 if (!tq.empty()) {
119 // Duration until next expiry. Mitigate scheduler latency by preempting the
120 // high-priority timer and busy-waiting for 200us ahead of timer expiry.
121 next = min(next, tq.front().expiry() - 200us);
122 }
123 }
124 {
125 const auto& tq = tqs_[Low];
126 if (!tq.empty()) {
127 // Duration until next expiry.
128 next = min(next, tq.front().expiry());
129 }
130 }
131 return next;
132}

134int Reactor::dispatch(CyclTime now, Event* buf, int size)
135{
136 int work{0};
137 for (int i{0}; i < size; ++i) {
138
139 auto& ev = buf[i];
140 const auto fd = epoll_.fd(ev);
141 if (fd == notify_.fd()) {
142 notify_.read();
143 continue;
144 }
145 const auto& ref = data_[fd];
146 if (!ref.slot) {
147 // Ignore timerfd.
148 continue;
149 }
150
151 const auto sid = epoll_.sid(ev);
152 // Skip this socket if it was modified after the call to wait().
153 if (ref.sid > sid) {
154 continue;
155 }
156 // Apply the interest events to filter-out any events that the user may have removed from
157 // the events since the call to wait() was made. This would typically happen via a reentrant
158 // call into the reactor from an event-handler. N.B. EpollErr and EpollHup are always
159 // reported if they occur, regardless of whether they are specified in events.
160 const auto events = ev.events & (ref.events | EpollErr | EpollHup);
161 if (!events) {
162 continue;
163 }
164
165 try {
166 ref.slot(now, fd, events);
167 } catch (const std::exception& e) {
168 TOOLBOX_ERROR << "exception in i/o event handler: " << e.what();
169 }
170 ++work;
171 }
172 return work;
173}

176{
