29int dispatch_low_priority_timers(CyclTime now, TimerQueue& tq, bool idle_cycle)
30{
31 int work_done = 0;
32 if (idle_cycle) {
33 work_done = tq.dispatch(now, 1);
34 }
35 else if (!tq.empty()) {
36 // actively execute low priority timers if they've been delayed by 100ms or more.
37 if ((now.mono_time() - tq.front().expiry()) > 100ms) {
38 work_done = tq.dispatch(now, 1);
39 }
40 }
41 return work_done;
42}

72int Reactor::poll(CyclTime now, Duration timeout)
73{
74 enum { High = 0, Low = 1 };
75 using namespace chrono;
76
77 // If timeout is zero then the wait_until time should also be zero to signify no wait.
78 MonoTime wait_until{};
79 if (!is_zero(timeout) && end_of_cycle_no_wait_hooks.empty()) {
80 const MonoTime next
81 = next_expiry(timeout == NoTimeout ? MonoClock::max() : now.mono_time() + timeout);
82 if (next > now.mono_time()) {
83 wait_until = next;
84 }
85 }
86 // TODO: consider using a dynamic buffer that scales with increased demand.
87 Event buf[MaxEvents];
88
89 int n;
90 error_code ec;
91 if (wait_until < MonoClock::max()) {
92 // The wait function will not block if time is zero.
93 n = epoll_.wait(buf, MaxEvents, wait_until, ec);
94 } else {
95 // Block indefinitely.
96 n = epoll_.wait(buf, MaxEvents, ec);
97 }
98 // Update cycle time after epoll() returns.
99 now = CyclTime::now();
100 if (ec) {
101 if (ec.value() != EINTR) {
102 throw system_error{ec};
103 }
104 return 0;
105 }
106 int work{0};
107 TOOLBOX_PROBE_SCOPED(reactor, dispatch, work);
108 // High priority timers.
109 work = tqs_[High].dispatch(now);
110 // I/O events.
111 work += dispatch(now, buf, n, Priority::High);
112 work += dispatch(now, buf, n, Priority::Low);
113 // Low priority timers (typically only dispatched during empty cycles).
114 work += dispatch_low_priority_timers(now, tqs_[Low], work == 0);
115 // End of cycle hooks.
116 if (work > 0) {
117 io::dispatch(now, end_of_event_dispatch_hooks_);
118 }
119 io::dispatch(now, end_of_cycle_no_wait_hooks);
120 return work;
121}

130MonoTime Reactor::next_expiry(MonoTime next) const
131{
132 enum { High = 0, Low = 1 };
133 using namespace chrono;
134 {
135 const auto& tq = tqs_[High];
136 if (!tq.empty()) {
137 // Duration until next expiry. Mitigate scheduler latency by preempting the
138 // high-priority timer and busy-waiting for 200us ahead of timer expiry.
139 next = min(next, tq.front().expiry() - 200us);
140 }
141 }
142 {
143 const auto& tq = tqs_[Low];
144 if (!tq.empty()) {
145 // Duration until next expiry.
146 next = min(next, tq.front().expiry());
147 }
148 }
149 return next;
150}

152int Reactor::dispatch(CyclTime now, Event* buf, int size, Priority priority)
153{
154 int work{0};
155 for (int i{0}; i < size; ++i) {
156
157 auto& ev = buf[i];
158 const auto fd = epoll_.fd(ev);
159 const auto& ref = data_[fd];
160
161 if (ref.priority != priority) {
162 continue;
163 }
164
165 if (fd == notify_.fd()) {
166 notify_.read();
167 continue;
168 }
169
170 if (!ref.slot) {
171 // Ignore timerfd.
172 continue;
173 }
174
175 const auto sid = epoll_.sid(ev);
176 // Skip this socket if it was modified after the call to wait().
177 if (ref.sid > sid) {
178 continue;
179 }
180 // Apply the interest events to filter-out any events that the user may have removed from
181 // the events since the call to wait() was made. This would typically happen via a reentrant
182 // call into the reactor from an event-handler. N.B. EpollErr and EpollHup are always
183 // reported if they occur, regardless of whether they are specified in events.
184 const auto events = ev.events & (ref.events | EpollErr | EpollHup);
185 if (!events) {
186 continue;
187 }
188
189 try {
190 ref.slot(now, fd, events);
191 } catch (const std::exception& e) {
192 TOOLBOX_ERROR << "exception in i/o event handler: " << e.what();
193 }
194 ++work;
195 }
196 return work;
197}
