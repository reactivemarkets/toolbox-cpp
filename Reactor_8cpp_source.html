42Reactor::Handle Reactor::subscribe(int fd, unsigned events, IoSlot slot)
43{
44 assert(fd >= 0);
45 assert(slot);
46 if (fd >= static_cast<int>(data_.size())) {
47 data_.resize(fd + 1);
48 }
49 auto& ref = data_[fd];
50 epoll_.add(fd, ++ref.sid, events);
51 ref.events = events;
52 ref.slot = slot;

56int Reactor::poll(CyclTime now, Duration timeout)
57{
58 enum { High = 0, Low = 1 };
59 using namespace chrono;
60
61 // If timeout is zero then the wait_until time should also be zero to signify no wait.
62 MonoTime wait_until{};
63 if (!is_zero(timeout) && hooks_.empty()) {
64 const MonoTime next
65 = next_expiry(timeout == NoTimeout ? MonoClock::max() : now.mono_time() + timeout);
66 if (next > now.mono_time()) {
67 wait_until = next;
68 }
69 }
70 // TODO: consider using a dynamic buffer that scales with increased demand.
71 Event buf[MaxEvents];
72
73 int n;
74 error_code ec;
75 if (wait_until < MonoClock::max()) {
76 // The wait function will not block if time is zero.
77 n = epoll_.wait(buf, MaxEvents, wait_until, ec);
78 } else {
79 // Block indefinitely.
80 n = epoll_.wait(buf, MaxEvents, ec);
81 }
82 if (ec) {
83 if (ec.value() != EINTR) {
84 throw system_error{ec};
85 }
86 return 0;
87 }
88 // If the epoller call was a blocking call, then acquire the current time.
89 if (!is_zero(wait_until)) {
90 now = CyclTime::now();
91 }
92 int work{0};
93 TOOLBOX_PROBE_SCOPED(reactor, dispatch, work);
94 // High priority timers.
95 work = tqs_[High].dispatch(now);
96 // I/O events.
97 work += dispatch(now, buf, n);
98 // Low priority timers are only dispatched during empty cycles.
99 if (work == 0) {
100 work += tqs_[Low].dispatch(now);
101 }
102 // End of cycle hooks.
103 io::dispatch(now, hooks_);
104 return work;
105}

114MonoTime Reactor::next_expiry(MonoTime next) const
115{
116 enum { High = 0, Low = 1 };
117 using namespace chrono;
118 {
119 const auto& tq = tqs_[High];
120 if (!tq.empty()) {
121 // Duration until next expiry. Mitigate scheduler latency by preempting the
122 // high-priority timer and busy-waiting for 200us ahead of timer expiry.
123 next = min(next, tq.front().expiry() - 200us);
124 }
125 }
126 {
127 const auto& tq = tqs_[Low];
128 if (!tq.empty()) {
129 // Duration until next expiry.
130 next = min(next, tq.front().expiry());
131 }
132 }
133 return next;
134}

136int Reactor::dispatch(CyclTime now, Event* buf, int size)
137{
138 int work{0};
139 for (int i{0}; i < size; ++i) {
140
141 auto& ev = buf[i];
142 const auto fd = epoll_.fd(ev);
143 if (fd == notify_.fd()) {
144 notify_.read();
145 continue;
146 }
147 const auto& ref = data_[fd];
148 if (!ref.slot) {
149 // Ignore timerfd.
150 continue;
151 }
152
153 const auto sid = epoll_.sid(ev);
154 // Skip this socket if it was modified after the call to wait().
155 if (ref.sid > sid) {
156 continue;
157 }
158 // Apply the interest events to filter-out any events that the user may have removed from
159 // the events since the call to wait() was made. This would typically happen via a reentrant
160 // call into the reactor from an event-handler. N.B. EpollErr and EpollHup are always
161 // reported if they occur, regardless of whether they are specified in events.
162 const auto events = ev.events & (ref.events | EpollErr | EpollHup);
163 if (!events) {
164 continue;
165 }
166
167 try {
168 ref.slot(now, fd, events);
169 } catch (const std::exception& e) {
170 TOOLBOX_ERROR << "exception in i/o event handler: " << e.what();
171 }
172 ++work;
173 }
174 return work;
175}
