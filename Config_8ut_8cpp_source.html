128BOOST_AUTO_TEST_CASE(ConfigOverrideCase)
129{
130 const string text{R"(
131foo=101
132bar=202
133
134[session]
135
136bar=303
137baz=404
138)"};
139
140 istringstream is{text};
141 string next;
142
143 Config parent;
144 parent.read_section(is, next);
145
146 BOOST_TEST(parent.size() == 2U);
147 BOOST_TEST(parent.get<int>("foo", 0) == 101);
148 BOOST_TEST(parent.get<int>("bar", 0) == 202);
149 BOOST_TEST(next == "session");
150 BOOST_TEST(!is.fail());
151
152 // Verify that getter with nullptr default compiles.
153 BOOST_TEST(parent.get("foo", nullptr));
154
155 // Conversion from internal std::string to std::string_view is a special case.
156 BOOST_TEST(parent.get<string_view>("foo") == "101"sv);
157
158 Config child;
159 child.read_section(is, next);
160 child.set_parent(parent);
161
162 BOOST_TEST(child.size() == 2U);
163 BOOST_TEST(child.get<int>("foo", 0) == 101);
164 BOOST_TEST(child.get<int>("bar", 0) == 303);
165 BOOST_TEST(child.get<int>("baz", 0) == 404);
166 BOOST_TEST(next.empty());
167 BOOST_TEST(is.eof());
168
169 BOOST_CHECK_THROW(child.get("bad"), runtime_error);
170 BOOST_CHECK_THROW(child.get<int>("bad"), runtime_error);
171}
