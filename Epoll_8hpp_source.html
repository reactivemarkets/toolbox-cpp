195  int wait(Event buf[], std::size_t size, std::error_code& ec) noexcept
196  {
197  MonoTime timeout{};
198  // Only set the timer if it has changed.
199  if (timeout != timeout_) {
200  // A zero timeout will disarm the timer.
201  tfd_.set_time(0, timeout, ec);
202  if (ec) {
203  return 0;
204  }
205  timeout_ = timeout;
206  }
207  return os::epoll_wait(*epfd_, buf, size, -1, ec);
208  }
213  int wait(Event buf[], std::size_t size, MonoTime timeout, std::error_code& ec) noexcept
214  {
215  // Only set the timer if it has changed.
216  if (timeout != timeout_) {
217  // A zero timeout will disarm the timer.
218  tfd_.set_time(0, timeout, ec);
219  if (ec) {
220  return 0;
221  }
222  timeout_ = timeout;
223  }
224  // Do not block if timer is zero.
225  return os::epoll_wait(*epfd_, buf, size, is_zero(timeout) ? 0 : -1, ec);
226  }
227  void add(int fd, int sid, unsigned events)
228  {
229  Event ev;
230  mod(ev, fd, sid, events);
231  os::epoll_ctl(*epfd_, EPOLL_CTL_ADD, fd, ev);
232  }
233  void del(int fd) noexcept
234  {
235  // In kernel versions before 2.6.9, the EPOLL_CTL_DEL operation required a non-null pointer
236  // in event, even though this argument is ignored.
237  Event ev{};
238  std::error_code ec;
239  os::epoll_ctl(*epfd_, EPOLL_CTL_DEL, fd, ev, ec);
240  }
241  void mod(int fd, int sid, unsigned events, std::error_code& ec) noexcept
242  {
243  Event ev;
244  mod(ev, fd, sid, events);
245  os::epoll_ctl(*epfd_, EPOLL_CTL_MOD, fd, ev, ec);
246  }
247  void mod(int fd, int sid, unsigned events)
248  {
249  Event ev;
250  mod(ev, fd, sid, events);
251  os::epoll_ctl(*epfd_, EPOLL_CTL_MOD, fd, ev);
252  }
253 
254  private:
255  static void mod(Event& ev, int fd, int sid, unsigned events) noexcept
256  {
257  ev.events = events;
258  ev.data.u64 = static_cast<std::uint64_t>(sid) << 32 | fd;
259  }
260  FileHandle epfd_;

262  MonoTime timeout_{};
