158
159 static constexpr int fd(const Event& ev) noexcept
160 {
161 return static_cast<int>(ev.data.u64 & 0xffffffff);
162 }
163 static constexpr int sid(const Event& ev) noexcept
164 {
165 return static_cast<int>(ev.data.u64 >> 32);
166 }
167 explicit Epoll(int flags = 0)
168 : epfd_{os::epoll_create1(flags)}
169 , tfd_{TFD_NONBLOCK}
170 {
171 add(tfd_.fd(), 0, EpollIn);
172 }
173 ~Epoll() { del(tfd_.fd()); }

175 // Copy.
176 Epoll(const Epoll&) = delete;
177 Epoll& operator=(const Epoll&) = delete;
178
179 // Move.
180 Epoll(Epoll&&) noexcept = default;
181 Epoll& operator=(Epoll&&) noexcept = default;
182
186 int timer_fd() const noexcept { return tfd_.fd(); }
187
188 void swap(Epoll& rhs) noexcept { std::swap(epfd_, rhs.epfd_); }
189
191 int wait(Event buf[], std::size_t size, std::error_code& ec) noexcept
192 {
193 MonoTime timeout{};
194 // Only set the timer if it has changed.
195 if (timeout != timeout_) {
196 // A zero timeout will disarm the timer.
197 tfd_.set_time(0, timeout, ec);
198 if (ec) {
199 return 0;
200 }
201 timeout_ = timeout;
202 }
203 return os::epoll_wait(*epfd_, buf, size, -1, ec);
204 }
209 int wait(Event buf[], std::size_t size, MonoTime timeout, std::error_code& ec) noexcept
210 {
211 // Only set the timer if it has changed.
212 if (timeout != timeout_) {
213 // A zero timeout will disarm the timer.
214 tfd_.set_time(0, timeout, ec);
215 if (ec) {
216 return 0;
217 }
218 timeout_ = timeout;
219 }
220 // Do not block if timer is zero.
221 return os::epoll_wait(*epfd_, buf, size, is_zero(timeout) ? 0 : -1, ec);
222 }
223 void add(int fd, int sid, unsigned events)
224 {
225 Event ev;
226 mod(ev, fd, sid, events);
227 os::epoll_ctl(*epfd_, EPOLL_CTL_ADD, fd, ev);
228 }
229 void del(int fd) noexcept
230 {
231 // In kernel versions before 2.6.9, the EPOLL_CTL_DEL operation required a non-null pointer
232 // in event, even though this argument is ignored.
233 Event ev{};
234 std::error_code ec;
235 os::epoll_ctl(*epfd_, EPOLL_CTL_DEL, fd, ev, ec);
236 }
237 void mod(int fd, int sid, unsigned events, std::error_code& ec) noexcept
238 {
239 Event ev;
240 mod(ev, fd, sid, events);
241 os::epoll_ctl(*epfd_, EPOLL_CTL_MOD, fd, ev, ec);
242 }
243 void mod(int fd, int sid, unsigned events)
244 {
245 Event ev;
246 mod(ev, fd, sid, events);
247 os::epoll_ctl(*epfd_, EPOLL_CTL_MOD, fd, ev);
248 }
249
250 private:
251 static void mod(Event& ev, int fd, int sid, unsigned events) noexcept
252 {
253 ev.events = events;
254 ev.data.u64 = static_cast<std::uint64_t>(sid) << 32 | fd;
255 }
256 FileHandle epfd_;
257 TimerFd<MonoClock> tfd_;
258 MonoTime timeout_{};
259};
