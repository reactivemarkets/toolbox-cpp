37 int32_t buckets_needed_to_cover_value(int64_t value, int64_t sub_bucket_count,
38  int32_t unit_magnitude) noexcept
39 {
40  int64_t smallest_untrackable_value{sub_bucket_count << unit_magnitude};
41  int32_t buckets_needed{1};
42  while (smallest_untrackable_value <= value) {
43  if (smallest_untrackable_value > numeric_limits<int64_t>::max() / 2) {
44  return buckets_needed + 1;
45  }
46  smallest_untrackable_value <<= 1;
47  ++buckets_needed;
48  }
49  return buckets_needed;
50 }

58 int32_t get_sub_bucket_half_count_magnitude(int64_t significant_figures) noexcept
59 {
60  const double largest_value_with_single_unit_resolution{pow(10, significant_figures) * 2.0};
61  const int32_t sub_bucket_count_magnitude
62  = ceil(log(largest_value_with_single_unit_resolution) / log(2));
63  return sub_bucket_count_magnitude > 1 ? sub_bucket_count_magnitude - 1 : 0;
64 }

68 HdrBucketConfig::HdrBucketConfig(int64_t lowest_trackable_value, int64_t highest_trackable_value,
69  int32_t significant_figures)
70 : lowest_trackable_value{lowest_trackable_value}
71 , highest_trackable_value{highest_trackable_value}
72 , significant_figures{significant_figures}
73 , unit_magnitude(floor(log(lowest_trackable_value) / log(2)))
74 , sub_bucket_half_count_magnitude{get_sub_bucket_half_count_magnitude(significant_figures)}
75 , sub_bucket_count(pow(2, sub_bucket_half_count_magnitude + 1))
76 , sub_bucket_half_count{sub_bucket_count / 2}
77 , sub_bucket_mask{int64_t{sub_bucket_count - 1} << unit_magnitude}
78 , bucket_count{buckets_needed_to_cover_value(highest_trackable_value, sub_bucket_count,
79  unit_magnitude)}
80 , counts_len{(bucket_count + 1) * (sub_bucket_count / 2)}
81 {
82  if (lowest_trackable_value < 1) {
83  throw invalid_argument{"min value must be greater than zero"};
84  }
85  if (significant_figures < 1 || significant_figures > 5) {
86  throw invalid_argument{"significant figures must be between 1 and 5"};
87  }
88  if (lowest_trackable_value * 2 > highest_trackable_value) {
89  throw invalid_argument{"highest trackable value too small"};
90  }
91  if (unit_magnitude + sub_bucket_half_count_magnitude > 61) {
92  throw invalid_argument{"invalid magnitude"};
93  }
94 }

164 int64_t HdrHistogram::value_at_index(int32_t index) const noexcept
165 {
166  int32_t bucket_index{(index >> sub_bucket_half_count_magnitude_) - 1};
167  int32_t sub_bucket_index{(index & (sub_bucket_half_count_ - 1)) + sub_bucket_half_count_};
168 
169  if (bucket_index < 0) {
170  sub_bucket_index -= sub_bucket_half_count_;
171  bucket_index = 0;
172  }
173  return value_from_index(bucket_index, sub_bucket_index, unit_magnitude_);
174 }

227 int32_t HdrHistogram::normalize_index(int32_t index) const noexcept
228 {
229  if (normalizing_index_offset_ == 0) {
230  return index;
231  }
232 
233  int32_t normalized_index{index - normalizing_index_offset_};
234 
235  int32_t adjustment{0};
236  if (normalized_index < 0) {
237  adjustment = counts_len();
238  } else if (normalized_index >= counts_len()) {
239  adjustment = -counts_len();
240  }
241  return normalized_index + adjustment;
242 }

251 int32_t HdrHistogram::counts_index(int32_t bucket_index, int32_t sub_bucket_index) const noexcept
252 {
253  // Calculate the index for the first entry in the bucket.
254  // The following is the equivalent of ((bucket_index + 1) * subBucketHalfCount)).
255  const int32_t bucket_base_index{(bucket_index + 1) << sub_bucket_half_count_magnitude_};
256  // Calculate the offset in the bucket.
257  const int32_t offset_in_bucket{sub_bucket_index - sub_bucket_half_count_};
258  // The following is the equivalent of
259  // (sub_bucket_index - subBucketHalfCount) + bucketBaseIndex).
260  return bucket_base_index + offset_in_bucket;
261 }
