26 int hex_digits(int64_t i) noexcept
27 {
28  int n{0};
29  if (i & 0xffffffff00000000) {
30  n += 8;
31  i >>= 32;
32  }
33  if (i & 0xffff0000) {
34  n += 4;
35  i >>= 16;
36  }
37  if (i & 0xff00) {
38  n += 2;
39  i >>= 8;
40  }
41  if (i & 0xf0) {
42  n += 2;
43  } else if (i & 0x0f) {
44  ++n;
45  }
46  return n;
47 }
