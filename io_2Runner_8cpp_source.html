27 void run_reactor(Reactor& r, long busy_cycles, ThreadConfig config, const std::atomic<bool>& stop)
28 {
29  sig_block_all();
30  try {
31  set_thread_attrs(config);
32  TOOLBOX_NOTICE << "started " << config.name << " thread";
33  long i{0};
34  while (!stop.load(std::memory_order_acquire)) {
35  // Busy-wait for "busy cycles" after work was done.
36  if (r.poll(CyclTime::now(), i++ < busy_cycles ? 0s : NoTimeout) > 0) {
37  // Reset counter when work has been done.
38  i = 0;
39  }
40  }
41  } catch (const std::exception& e) {
42  TOOLBOX_CRIT << "exception: " << e.what();
43  kill(getpid(), SIGTERM);
44  }
45  TOOLBOX_NOTICE << "stopping " << config.name << " thread";
46 }
47 } // namespace
