34cpu_set_t parse_cpu_set(string_view s) noexcept
35{
36 cpu_set_t bs;
37 CPU_ZERO(&bs);
38
39 Tokeniser toks{s, ","sv};
40 while (!toks.empty()) {
41 auto [i, j] = split_range(toks.top());
42 for (; i <= j; ++i) {
43 CPU_SET(i, &bs);
44 }
45 toks.pop();
46 }
47 return bs;
48}

50int parse_sched_policy(string_view s)
51{
52 int policy;
53 if (s == "other"sv) {
54 policy = SCHED_OTHER;
55 } else if (s == "fifo"sv) {
56 policy = SCHED_FIFO;
57 } else if (s == "rr"sv) {
58 policy = SCHED_RR;
59 } else if (s == "batch"sv) {
60 policy = SCHED_BATCH;
61 } else if (s == "idle"sv) {
62 policy = SCHED_IDLE;
63 } else {
64 throw invalid_argument{"invalid sched policy"};
65 }
66 return policy;
67}

69void set_thread_attrs(const ThreadConfig& config)
70{
71 const auto tid = pthread_self();
72 if (!config.name.empty()) {
73 if (const auto err = pthread_setname_np(tid, config.name.c_str()); err != 0) {
74 throw system_error{make_error(err), "pthread_setname_np"};
75 }
76 }
77 if (!config.affinity.empty()) {
78 const auto bs = parse_cpu_set(config.affinity);
79 if (const auto err = pthread_setaffinity_np(tid, sizeof(bs), &bs); err != 0) {
80 throw system_error{make_error(err), "pthread_setaffinity_np"};
81 }
82 }
83 if (!config.sched_policy.empty()) {
84 const auto policy = parse_sched_policy(config.sched_policy);
85 const auto priority = sched_get_priority_max(policy);
86 if (priority == -1) {
87 throw system_error{make_error(priority), "sched_get_priority_max"};
88 }
89 sched_param param{.sched_priority = priority};
90 if (const auto err = pthread_setschedparam(tid, policy, &param); err != 0) {
91 throw system_error{make_error(err), "pthread_setschedparam"};
92 }
93 }
94}
