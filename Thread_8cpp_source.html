34 cpu_set_t parse_cpu_set(string_view s) noexcept
35 {
36  cpu_set_t bs;
37  CPU_ZERO(&bs);
38 
39  Tokeniser toks{s, ","sv};
40  while (!toks.empty()) {
41  auto [i, j] = split_range(toks.top());
42  for (; i <= j; ++i) {
43  CPU_SET(i, &bs);
44  }
45  toks.pop();
46  }
47  return bs;

70 {
71  const auto tid = pthread_self();

74  throw system_error{make_error(err), "pthread_setname_np"};

79  if (const auto err = pthread_setaffinity_np(tid, sizeof(bs), &bs); err != 0) {
80  throw system_error{make_error(err), "pthread_setaffinity_np"};
81  }
82  }
83  if (!config.sched_policy.empty()) {

85  const auto priority = sched_get_priority_max(policy);
