29 struct ThreadSafePolicy {
30  using Type = std::atomic<int>;
31  static void acquire() noexcept { std::atomic_thread_fence(std::memory_order_acquire); }
32  static int fetch_add(Type& ref_count) noexcept
33  {
34  return ref_count.fetch_add(1, std::memory_order_relaxed);
35  }
36  static int fetch_sub(Type& ref_count) noexcept
37  {
38  return ref_count.fetch_sub(1, std::memory_order_release);
39  }
40  static int load(const Type& ref_count) noexcept
41  {

69  void add_ref() const noexcept { PolicyT::fetch_add(refs_); }
70  void release() const noexcept
71  {
