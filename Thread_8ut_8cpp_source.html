25uint32_t to_bitset(const cpu_set_t& cpuset) noexcept
26{
27 uint32_t bs{0};
28 for (int i{0}; i < 32; ++i) {
29 if (CPU_ISSET(i, &cpuset)) {
30 bs |= (1 << i);
31 }
32 }
33 return bs;
34}
