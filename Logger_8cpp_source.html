57 class StdLogger final : public Logger {
58  void do_write_log(WallTime ts, LogLevel level, LogMsgPtr&& msg, size_t size) noexcept override
59  {
60  const auto t = WallClock::to_time_t(ts);
61  const auto ms = ms_since_epoch(ts);
62 
63  struct tm tm; // NOLINT(hicpp-member-init)
64  localtime_r(&t, &tm);
65 
66  // The following format has an upper-bound of 42 characters:
67  // "%b %d %H:%M:%S.%03d %-7s [%d]: "
68  //
69  // Example:
70  // Mar 14 00:00:00.000 WARNING [0123456789]: msg...
71  // <---------------------------------------->
72  char head[42 + 1];
73  size_t hlen = strftime(head, sizeof(head), "%b %d %H:%M:%S", &tm);
74  hlen += sprintf(head + hlen, ".%03d %-7s [%d]: ", static_cast<int>(ms % 1000),
75  log_label(level), static_cast<int>(gettid()));
76  char tail{'\n'};
77  iovec iov[] = {
78  {head, hlen}, //
79  {msg.get(), size}, //
80  {&tail, 1} //
81  };
82 
83  int fd{level > LogLevel::Error ? STDOUT_FILENO : STDERR_FILENO};
84  // The following lock was required to avoid interleaving.
85  lock_guard<mutex> lock{mutex_};
86  // Best effort given that this is the logger.
87 #pragma GCC diagnostic push
88 #pragma GCC diagnostic ignored "-Wunused-result"
89  writev(fd, iov, sizeof(iov) / sizeof(iov[0]));
90 #pragma GCC diagnostic pop
91  }
92  mutex mutex_;
93 } std_logger_;

95 class SysLogger final : public Logger {
96  void do_write_log(WallTime ts, LogLevel level, LogMsgPtr&& msg, size_t size) noexcept override
97  {
98  int prio;
99  switch (level) {
100  case LogLevel::Crit:
101  prio = LOG_CRIT;
102  break;
103  case LogLevel::Error:
104  prio = LOG_ERR;
105  break;
106  case LogLevel::Warning:
107  prio = LOG_WARNING;
108  break;
109  case LogLevel::Notice:
110  prio = LOG_NOTICE;
111  break;
112  case LogLevel::Info:
113  prio = LOG_INFO;
114  break;
115  default:
116  prio = LOG_DEBUG;
117  }
118  syslog(prio, "%.*s", static_cast<int>(size), static_cast<const char*>(msg.get()));
119  }
120 } sys_logger_;
