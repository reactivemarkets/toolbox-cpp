29BOOST_AUTO_TEST_CASE(GetUnixAddrInfoCase)
30{
31 auto path = "/tmp/foo.sock"s;
32 const auto ai = get_unix_addrinfo(path, SOCK_DGRAM);
33 BOOST_TEST(ai->ai_flags == 0);
34 BOOST_TEST(ai->ai_family == AF_UNIX);
35 BOOST_TEST(ai->ai_socktype == SOCK_DGRAM);
36 BOOST_TEST(ai->ai_protocol == 0);
37 BOOST_TEST(ai->ai_addrlen == offsetof(sockaddr_un, sun_path) + path.size() + 1);
38 BOOST_TEST(ai->ai_addr->sa_family == AF_UNIX);
39 BOOST_TEST(!ai->ai_canonname);
40 BOOST_TEST(!ai->ai_next);
41 BOOST_TEST(to_string(*ai->ai_addr) == path);

47BOOST_AUTO_TEST_CASE(AbstractUnixSocketCase)
48{
49 // From unix(7) man page
50 // If a bind(2) call specifies addrlen as sizeof(sa_family_t), or the SO_PASSCRED socket option
51 // was specified for a socket that was not explicitly bound to an address, then the socket is
52 // auto_bound to an abstract address. The address consists of a null byte followed by 5 bytes in
53 // the character set [0-9a-f]. Thus, there is a limit of 2^20 autobind addresses. (From Linux
54 // 2.1.15, when the autobind feature was added, 8 bytes were used, and the limit was thus 2^32
55 // autobind addresses. The change to 5 bytes came in Linux 2.3.15.)
56
57 FileHandle acc_sock, server_sock, client_sock;
58 BOOST_REQUIRE_NO_THROW(acc_sock = os::socket(AF_UNIX, SOCK_STREAM, 0));
59 sockaddr_un addr{.sun_family = AF_UNIX, .sun_path = {0}};
60 BOOST_REQUIRE_NO_THROW(
61 os::bind(acc_sock.get(), *reinterpret_cast<sockaddr*>(&addr), sizeof(addr.sun_family)));
62 // get the auto-bound address
63 StreamEndpoint auto_bound_endpoint;
64 BOOST_REQUIRE_NO_THROW(os::getsockname(acc_sock.get(), auto_bound_endpoint));
65 BOOST_CHECK_EQUAL(auto_bound_endpoint.size(), 8);
66 BOOST_REQUIRE_NO_THROW(os::listen(acc_sock.get(), 1));
67 // connect the client
68 BOOST_REQUIRE_NO_THROW(client_sock = os::socket(auto_bound_endpoint.protocol()));
69 BOOST_REQUIRE_NO_THROW(os::connect(client_sock.get(), auto_bound_endpoint));
70 // accept the connection
71 BOOST_REQUIRE_NO_THROW(server_sock = os::accept(acc_sock.get()));
72
73 std::string msg_sent = "Hello";
74 BOOST_REQUIRE_EQUAL(
75 ssize_t(msg_sent.size()),
76 os::send(client_sock.get(), msg_sent.data(), msg_sent.size(), MSG_DONTWAIT));
77 std::string msg_recv(msg_sent.size(), '\0');
78 BOOST_REQUIRE_NO_THROW(os::recv(server_sock.get(), const_cast<char*>(msg_recv.data()),
79 msg_sent.size(), MSG_DONTWAIT));
80 BOOST_CHECK_EQUAL(msg_sent, msg_recv);
81}
