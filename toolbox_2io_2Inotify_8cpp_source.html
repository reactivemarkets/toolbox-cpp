28void FileWatcher::on_inotify(CyclTime /*now*/, int fd, unsigned events)
29{
30 if (events & (EpollIn | EpollHup)) {
31 char buf[1024 * (sizeof(struct inotify_event) + NAME_MAX + 1)];
32 const auto size{os::read(fd, buf, sizeof(buf))};
33 if (size == 0) {
34 // FIXME: a zero return normally indicates end of file.
35 // Can this happen on an inotify descriptor?
36 // And if so, how should the application behave?
37 watch_dir_.reset();
38 sub_.reset();
39 return;
40 }
41 for (std::size_t i{0}; i < size;) {
42 inotify_event* event{reinterpret_cast<inotify_event*>(&buf[i])};
43 const Path file_name{event->name};
44 const auto it{slot_map_.find(file_name)};
45 if (it != slot_map_.end()) {
46 it->second(file_name, event->mask);
47 }
48 i += sizeof(struct inotify_event) + event->len;
49 }
50 }
51}
