31 class EchoConn {
32 
33  // Automatically unlink when object is destroyed.
34  using AutoUnlinkOption = boost::intrusive::link_mode<boost::intrusive::auto_unlink>;
35 
36  public:
37  EchoConn(CyclTime now, Reactor& r, IoSock&& sock, const StreamEndpoint& ep)
38  : sock_{move(sock)}
39  , ep_{ep}
40  {
41  sub_ = r.subscribe(sock_.get(), EpollIn, bind<&EchoConn::on_input>(this));
42  tmr_ = r.timer(now.mono_time(), PingInterval, Priority::Low,
43  bind<&EchoConn::on_timer>(this));
44  }
45  void dispose(CyclTime now) noexcept
46  {
47  TOOLBOX_INFO << "connection closed";
48  delete this;
49  }
50  boost::intrusive::list_member_hook<AutoUnlinkOption> list_hook;
51 
52  private:
53  ~EchoConn() = default;
54  void on_input(CyclTime now, int fd, unsigned events)
55  {
56  try {
57  if (events & (EpollIn | EpollHup)) {
58  const auto size = os::read(fd, buf_.prepare(2944));
59  if (size == 0) {
60  dispose(now);
61  return;
62  }
63  // Commit actual bytes read.
64  buf_.commit(size);
65 
66  // Parse each buffered line.
67  auto fn = [this](std::string_view line) {
68  ++count_;
69  // Echo bytes back to client.
70  TOOLBOX_INFO << "received: " << line;
71  };
72  buf_.consume(parse_line(buf_.str(), fn));
73  if (count_ == 5) {
74  dispose(now);
75  return;
76  }
77  }
78  } catch (const std::exception& e) {
79  TOOLBOX_ERROR << "exception on input: " << e.what();
80  dispose(now);
81  }
82  }
83  void on_timer(CyclTime now, Timer& tmr)
84  {
85  try {
86  if (sock_.send("ping\n", 5, 0) < 5) {
87  throw runtime_error{"partial write"};
88  }
89  } catch (const std::exception& e) {
90  TOOLBOX_ERROR << "exception on timer: " << e.what();
91  dispose(now);
92  }
93  }
94  IoSock sock_;
95  const StreamEndpoint ep_;
96  Reactor::Handle sub_;
97  Buffer buf_;
98  Timer tmr_;
99  int count_{0};
100 };

128  {

135  void on_sock_connect(CyclTime now, IoSock&& sock, const Endpoint& ep)
136  {
137  TOOLBOX_INFO << "connection opened: " << ep;
138  inprogress_ = false;
139 
140  // High performance TCP servers could use a custom allocator.
141  auto* const conn = new EchoConn{now, reactor_, move(sock), ep};
142  conn_list_.push_back(*conn);
143  }
144  void on_sock_connect_error(CyclTime now, const std::exception& e)
145  {
146  TOOLBOX_ERROR << "could not connect: " << e.what();
147  aifuture_ = resolver_.resolve(uri_, SOCK_STREAM);
148  inprogress_ = false;
149  }
150  void on_timer(CyclTime now, Timer& tmr)
151  {
152  if (!conn_list_.empty() || inprogress_) {
153  return;
154  }
155  if (aifuture_.valid()) {
156  if (!is_ready(aifuture_)) {
157  TOOLBOX_INFO << "address pending";
158  return;
159  }
160  try {
161  ep_ = get_endpoint<Endpoint>(aifuture_);
162  } catch (const std::exception& e) {
163  TOOLBOX_ERROR << "could not resolve address: " << e.what();
164  aifuture_ = resolver_.resolve(uri_, SOCK_STREAM);
165  return;
166  }
167  TOOLBOX_INFO << "address resolved: " << ep_;
168  }
169  TOOLBOX_INFO << "reconnecting";
170  if (!connect(now, reactor_, ep_)) {
171  inprogress_ = true;
172  }
173  }
174  Reactor& reactor_;
175  Resolver& resolver_;
176  const string uri_;
177  Timer tmr_;

179  Endpoint ep_;
