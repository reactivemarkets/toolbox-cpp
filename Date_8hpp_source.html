58 constexpr auto ymd_to_jd(int year, int mon, int mday) noexcept
59 {
60  // The formula given below was taken from the 1990 edition of the U.S. Naval Observatory's Almanac
61  // for Computers.
62  // See http://aa.usno.navy.mil/faq/docs/JD_Formula.php.
63 
64  const auto i = year;
65  const auto j = mon;
66  const auto k = mday;
67  return JDay{k - 32075 + 1461 * (i + 4800 + (j - 14) / 12) / 4
68  + 367 * (j - 2 - (j - 14) / 12 * 12) / 12
69  - 3 * ((i + 4900 + (j - 14) / 12) / 100) / 4};
70 }

83 constexpr auto jd_to_iso(JDay jd) noexcept
84 {
85  // The formula given above was taken from the 1990 edition of the U.S. Naval Observatory's
86  // Almanac for Computers.
87  // See http://aa.usno.navy.mil/faq/docs/JD_Formula.php.
88 
89  auto l = jd.count() + 68569;
90  const auto n = 4 * l / 146097;
91  l = l - (146097 * n + 3) / 4;
92  auto i = 4000 * (l + 1) / 1461001;
93  l = l - 1461 * i / 4 + 31;
94  auto j = 80 * l / 2447;
95  const auto k = l - 2447 * j / 80;
96  l = j / 11;
97  j = j + 2 - 12 * l;
98  i = 100 * (n - 49) + i + l;
99 
100  return IsoDate{i * 10000 + j * 100 + k};
101 }

128 constexpr WallTime jd_to_time(JDay jd) noexcept
129 {
130  // Julian day for January 1st, 1970.
131  const JDay jd_unix_epoc = 2440588_jd;
132  const std::int64_t ms_in_day = 24 * 60 * 60 * 1000;
133  // Add half day for 12pm.
134  return to_time<WallClock>(Millis{(jd - jd_unix_epoc).count() * ms_in_day + (ms_in_day >> 1)});
135 }

185 constexpr std::optional<WallTime> parse_time(std::string_view sv) noexcept
186 {
187  if (sv.size() < 8) {
188  // Date part is too short.
189  return {};
190  }
191  const IsoDate d{parse_date(sv.substr(0, 8))};
192  WallTime dt{};
193  if (d != 0_ymd) {
194  // Julian days start at noon.
195  dt = jd_to_time(iso_to_jd(d)) - 12h;
196  }
197  // If there is a time component.
198  if (sv.size() > 8) {
199  if (sv[8] != '-' && sv[8] != 'T') {
200  // Invalid delimiter.
201  return {};
202  }
203  const auto t = parse_time_only(sv.substr(9));
204  if (!t) {
205  // Invalid time.
206  return {};
207  }
208  dt += *t;
209  }
210  return dt;
211 }
