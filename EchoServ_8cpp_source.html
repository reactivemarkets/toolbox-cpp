31 class EchoConn {
32 
33  // Automatically unlink when object is destroyed.
34  using AutoUnlinkOption = boost::intrusive::link_mode<boost::intrusive::auto_unlink>;
35 
36  public:
37  template <typename EndpointT>
38  EchoConn(CyclTime now, Reactor& r, IoSock&& sock, const EndpointT& ep)
39  : reactor_{r}
40  , sock_{move(sock)}
41  , ep_{ep}
42  {
43  sub_ = r.subscribe(sock_.get(), EpollIn, bind<&EchoConn::on_input>(this));
44  tmr_ = r.timer(now.mono_time() + IdleTimeout, Priority::Low,
45  bind<&EchoConn::on_timer>(this));
46  }
47  void dispose(CyclTime now) noexcept
48  {
49  TOOLBOX_INFO << "connection closed";
50  delete this;
51  }
52  boost::intrusive::list_member_hook<AutoUnlinkOption> list_hook;
53 
54  private:
55  ~EchoConn() = default;
56  void on_input(CyclTime now, int fd, unsigned events)
57  {
58  try {
59  if (events & (EpollIn | EpollHup)) {
60  const auto size = os::read(fd, buf_.prepare(2944));
61  if (size == 0) {
62  dispose(now);
63  return;
64  }
65  // Commit actual bytes read.
66  buf_.commit(size);
67 
68  // Parse each buffered line.
69  auto fn = [fd](std::string_view line) {
70  // Echo bytes back to client.
71  std::string buf{line};
72  buf += '\n';
73  if (os::write(fd, {buf.data(), buf.size()}) < buf.size()) {
74  throw runtime_error{"partial write"};
75  }
76  };
77  buf_.consume(parse_line(buf_.str(), fn));
78 
79  // Reset timer.
80  tmr_.cancel();
81  tmr_ = reactor_.timer(now.mono_time() + IdleTimeout, Priority::Low,
82  bind<&EchoConn::on_timer>(this));
83  }
84  } catch (const std::exception& e) {
85  TOOLBOX_ERROR << "exception on input: " << e.what();
86  dispose(now);
87  }
88  }
89  void on_timer(CyclTime now, Timer& tmr)
90  {
91  TOOLBOX_INFO << "timeout";
92  dispose(now);
93  }
94  Reactor& reactor_;
95  IoSock sock_;
96  const StreamEndpoint ep_;
97  Reactor::Handle sub_;
98  Buffer buf_;
99  Timer tmr_;
100 };
