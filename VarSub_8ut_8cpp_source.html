27string apply_copy(FnT fn, string s)

33string get_var(const string& name)
34{
35 string val;
36 if (name == "FOO") {
37 val = "101";
38 } else if (name == "BAR") {
39 val = "202";
40 } else if (name == "BAZ") {
41 val = "FOO";
42 } else if (name == "QUX") {
43 val = "BAR";
44 } else if (name == "FOOBAR") {
