185  void on_io_event(CyclTime now, int fd, unsigned events)
186  {
187  auto lock = this->lock_this(now);
188  try {
189  if (events & (EpollIn | EpollHup)) {
190  if (!drain_input(now, fd)) {
191  this->dispose(now);
192  return;
193  }
194  }
195  // Do not attempt to flush the output buffer if it is empty or if we are still waiting
196  // for the socket to become writable.
197  if (out_.empty() || (write_blocked_ && !(events & EpollOut))) {
198  return;
199  }
200  flush_output(now);
201  } catch (const Exception&) {
202  // Do not call on_http_error() here, because it will have already been called in one of
203  // the noexcept parser callback functions.
204  } catch (const std::exception& e) {
205  app_.on_http_error(now, ep_, e, os_);
206  this->dispose(now);
207  }
208  }
209  bool drain_input(CyclTime now, int fd)
210  {
211  // Limit the number of reads to avoid starvation.
212  for (int i{0}; i < 4; ++i) {
213  std::error_code ec;
214  const auto buf = in_.prepare(2944);
215  const auto size = os::read(fd, buf, ec);
216  if (ec) {
217  // No data available in socket buffer.
218  if (ec == std::errc::operation_would_block) {
219  break;
220  }
221  throw std::system_error{ec, "read"};
222  }
223  if (size == 0) {
224  // N.B. the socket may still be writable if the peer has performed a shutdown on the
225  // write side of the socket only.
226  flush_input(now);
227  return false;
228  }
229  // Commit actual bytes read.
230  in_.commit(size);
231  // Assume that the TCP stream has been drained if we read less than the requested
232  // amount.
233  if (static_cast<size_t>(size) < buffer_size(buf)) {
234  break;
235  }
236  }
237  flush_input(now);
238  // Reset timer.
239  schedule_timeout(now);
240  return true;
241  }
242  void flush_input(CyclTime now) { in_.consume(parse(now, in_.data())); }
243  void flush_output(CyclTime now)
244  {
245  // Attempt to flush buffered data.
246  out_.consume(os::write(sock_.get(), out_.data()));
247  if (out_.empty()) {
248  if (!in_progress_ && !should_keep_alive()) {
249  this->dispose(now);
250  return;
251  }
252  if (write_blocked_) {
253  // Restore read-only state after the buffer has been drained.
254  sub_.set_events(EpollIn);
255  write_blocked_ = false;
256  }
257  } else if (!write_blocked_) {
258  // Set the state to read-write if the entire buffer could not be written.
259  sub_.set_events(EpollIn | EpollOut);
260  write_blocked_ = true;
261  }
262  }
263  void schedule_timeout(CyclTime now)
264  {
265  const auto timeout = std::chrono::ceil<Seconds>(now.mono_time() + IdleTimeout);
266  tmr_ = reactor_.timer(timeout, Priority::Low, bind<&BasicConn::on_timeout_timer>(this));
267  }
