34 template <typename FnT>
35 std::istream& parse_section(std::istream& is, FnT fn, std::string* name = nullptr)
36 {
37  std::string line;
38  while (std::getline(is, line)) {
39 
40  trim(line);
41 
42  // Ignore empty lines and comments.
43 
44  if (line.empty() || line[0] == '#') {
45  continue;
46  }
47 
48  if (line.front() == '[' && line.back() == ']') {
49  if (name) {
50  name->assign(line, 1, line.size() - 2);
51  trim(*name);
52  }
53  return is;
54  }
55 
56  auto [key, val] = split_pair(line, '=');
57  rtrim(key);
58  ltrim(val);
59 
60  fn(key, val);
61  }
62  if (name) {
63  name->clear();
64  }
65  return is;
66 }

92  {
93  auto it = map_.find(key);
94  if (it != map_.end()) {
95  if constexpr (std::is_same_v<ValueT, std::string_view>) {
96  // Explicitly allow conversion to std::string_view in this case, because we know
97  // that the argument is not a temporary.
98  return from_string<ValueT>(std::string_view{it->second});
99  } else if constexpr (std::is_enum_v<ValueT>) {
100  return ValueT{from_string<std::underlying_type_t<ValueT>>(it->second)};
101  } else {
102  return from_string<ValueT>(it->second);
103  }
104  }
