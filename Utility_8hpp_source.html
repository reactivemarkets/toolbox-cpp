50 template <typename ValueT>
51 constexpr ValueT ston(std::string_view sv) noexcept
52 {
53  if constexpr (std::is_same_v<ValueT, double>) {
54  return stod(sv);
55  } else if constexpr (std::is_integral_v<ValueT>) {
56  auto it = sv.begin(), end = sv.end();
57  if (it == end) {
58  return 0;
59  }
60  bool neg{false};
61  if constexpr (std::is_signed_v<ValueT>) {
62  // Handle sign.
63  if (*it == '-') {
64  if (++it == end) {
65  return 0;
66  }
67  neg = true;
68  }
69  }
70  std::uint64_t n{0};
71  if (isdigit(*it)) {
72  n = *it++ - '0';
73  while (it != end && isdigit(*it)) {
74  n *= 10;
75  n += *it++ - '0';
76  }
77  }
78  return neg ? -n : n;
79  } else {
80  static_assert(AlwaysFalse<ValueT>::value);
81  }
82 }
83 static_assert(ston<int>(std::string_view{"-123"}) == -123);
