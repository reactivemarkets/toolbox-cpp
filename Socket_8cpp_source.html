35AddrInfoPtr get_unix_addrinfo(string_view path, int type)
36{
37 auto sun = make_unique<sockaddr_un>();
38 sun->sun_family = AF_UNIX;
39 const socklen_t path_len = pstrcpy<'\0'>(sun->sun_path, path);
40 // Check that path has not exceeded max length.
41 if (path_len == sizeof(sun->sun_path)) {
42 throw invalid_argument{"invalid unix domain address"};
43 }
44
45 AddrInfoPtr ai{new addrinfo{}, free_unix_addrinfo};
46 ai->ai_family = AF_UNIX;
47 ai->ai_socktype = type;
48 // Size includes null terminator. See unix(7).
49 ai->ai_addrlen = offsetof(sockaddr_un, sun_path) + path_len + 1;
50 ai->ai_addr = reinterpret_cast<sockaddr*>(sun.release());
51 return ai;
52}
