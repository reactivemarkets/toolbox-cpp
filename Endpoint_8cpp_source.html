25 pair<string, string> split_ip_addr(const string& addr, char delim)
26 {
27  // Reverse find for compatibility with ipv6 address notation.
28  const auto pos = addr.find_last_of(delim);
29  string node, service;
30  if (pos == string::npos) {
31  node = addr;
32  } else {
33  node = addr.substr(0, pos);
34  service = addr.substr(pos + 1);
35  }
36  // Remove square braces arround ipv6 address.
37  if (node.size() >= 2 && node.front() == '[' && node.back() == ']') {
38  node = node.substr(1, node.size() - 2);
39  }
40  return {node, service};
41 }

57 AddrInfoPtr parse_endpoint(const string& uri, int type)
58 {
59  int family{-1}, protocol{0};
60  const auto [scheme, addr] = split_uri(uri);
61  if (scheme.empty()) {
62  family = AF_UNSPEC;
63  } else if (scheme == "ip4") {
64  family = AF_INET;
65  } else if (scheme == "ip6") {
66  family = AF_INET6;
67  } else if (scheme == "tcp4") {
68  if (type == SOCK_STREAM) {
69  family = AF_INET;
70  protocol = IPPROTO_TCP;
71  }
72  } else if (scheme == "tcp6") {
73  if (type == SOCK_STREAM) {
74  family = AF_INET6;
75  protocol = IPPROTO_TCP;
76  }
77  } else if (scheme == "udp4") {
78  if (type == SOCK_DGRAM) {
79  family = AF_INET;
80  protocol = IPPROTO_UDP;
81  }
82  } else if (scheme == "udp6") {
83  if (type == SOCK_DGRAM) {
84  family = AF_INET6;
85  protocol = IPPROTO_UDP;
86  }
87  } else if (scheme == "unix") {
88  return get_unix_addrinfo(addr, type);
89  }
90  if (family < 0) {
91  throw invalid_argument{"invalid uri: "s + uri};
92  }
93  auto [node, service] = split_ip_addr(addr, ':');
94  return os::getaddrinfo(!node.empty() ? node.c_str() : nullptr,
95  !service.empty() ? service.c_str() : nullptr, family, type, protocol);
96 }
