40 template <typename ValueT>
41 class MpmcQueue {
42  static_assert(std::is_trivially_copyable_v<ValueT>);
43 
44  public:
45  struct alignas(CacheLineSize) Elem {
46  std::int64_t seq;
47  ValueT val;
48  };
49  static_assert(std::is_trivially_copyable_v<Elem>);
50  struct alignas(CacheLineSize) Impl {
51  // Ensure that read and write positions are in different cache-lines.
52  std::int64_t rpos;
53  alignas(CacheLineSize) std::int64_t wpos;
54  alignas(CacheLineSize) Elem elems[];
55  };
56  static_assert(std::is_trivially_copyable_v<Impl>);
57  static_assert(sizeof(Impl) == 2 * CacheLineSize);
58  static_assert(offsetof(Impl, rpos) == 0 * CacheLineSize);
59  static_assert(offsetof(Impl, wpos) == 1 * CacheLineSize);
60  static_assert(offsetof(Impl, elems) == 2 * CacheLineSize);

62  constexpr MpmcQueue(std::nullptr_t = nullptr) noexcept {}
63  explicit MpmcQueue(std::size_t capacity)
64  : capacity_{next_pow2(capacity)}
65  , mask_{capacity_ - 1}
66  , mem_map_{os::mmap(nullptr, size(capacity_), PROT_READ | PROT_WRITE, MAP_ANON | MAP_PRIVATE,
67  -1, 0)}
68  , impl_{static_cast<Impl*>(mem_map_.get().data())}
69  {
70  assert(capacity >= 2);
71 
72  std::memset(impl_, 0, size(capacity_));
73  // Initialise sequence numbers.
74  for (std::int64_t i{0}; i < static_cast<std::int64_t>(capacity); ++i) {
75  __atomic_store_n(&impl_->elems[i].seq, i, __ATOMIC_RELAXED);
76  }
77  }

164  template <typename FnT>
165  bool read(FnT fn) noexcept
166  {
167  static_assert(std::is_nothrow_invocable_v<FnT, ValueT&&>);
168  auto rpos = __atomic_load_n(&impl_->rpos, __ATOMIC_RELAXED);
169  for (;;) {
170  auto& elem = impl_->elems[rpos & mask_];
171  const auto seq = __atomic_load_n(&elem.seq, __ATOMIC_ACQUIRE);
172  const auto diff = seq - (rpos + 1);
173  if (diff == 0) {
174  // The compare_exchange_weak function re-reads rpos on failure.
175  if (__atomic_compare_exchange_n(&impl_->rpos, &rpos, rpos + 1, true,
176  __ATOMIC_RELAXED, __ATOMIC_RELAXED)) {
177  fn(std::move(elem.val));
178  // Commit.
179  __atomic_store_n(&elem.seq, rpos + capacity_, __ATOMIC_RELEASE);
180  break;
181  }
182  // Continue.
183  } else if (diff < 0) {
184  return false;
185  } else {
186  rpos = __atomic_load_n(&impl_->rpos, __ATOMIC_RELAXED);
187  }
188  }
189  return true;
190  }
192  template <typename FnT>
193  bool write(FnT fn) noexcept
194  {
195  static_assert(std::is_nothrow_invocable_v<FnT, ValueT&>);
196  auto wpos = __atomic_load_n(&impl_->wpos, __ATOMIC_RELAXED);
197  for (;;) {
198  auto& elem = impl_->elems[wpos & mask_];
199  const auto seq = __atomic_load_n(&elem.seq, __ATOMIC_ACQUIRE);
200  const auto diff = seq - wpos;
201  if (diff == 0) {
202  // The compare_exchange_weak function re-reads wpos on failure.
203  if (__atomic_compare_exchange_n(&impl_->wpos, &wpos, wpos + 1, true,
204  __ATOMIC_RELAXED, __ATOMIC_RELAXED)) {
205  fn(elem.val);
206  // Commit.
207  __atomic_store_n(&elem.seq, wpos + 1, __ATOMIC_RELEASE);
208  break;
209  }
210  // Continue.
211  } else if (diff < 0) {
212  return false;
213  } else {
214  wpos = __atomic_load_n(&impl_->wpos, __ATOMIC_RELAXED);
215  }
216  }
217  return true;
218  }
220  bool pop(ValueT& val) noexcept
221  {
222  static_assert(std::is_nothrow_move_assignable_v<ValueT>);
223  return read([&val](ValueT && ref) noexcept { val = std::move(ref); });
224  }
226  bool push(const ValueT& val) noexcept
227  {
228  static_assert(std::is_nothrow_copy_assignable_v<ValueT>);
229  return write([&val](ValueT & ref) noexcept { ref = val; });
230  }
232  bool push(ValueT&& val) noexcept
233  {
234  static_assert(std::is_nothrow_move_assignable_v<ValueT>);
235  return write([&val](ValueT & ref) noexcept { ref = std::move(val); });
236  }

238  private:
239  static constexpr std::size_t capacity(std::size_t size) noexcept
240  {

242  }
243  static constexpr std::size_t size(std::size_t capacity) noexcept
244  {
245  return sizeof(Impl) + capacity * sizeof(Elem);
246  }

255 template <typename ValueT>
256 void create_mpmc_queue(FileHandle& fh, std::size_t capacity)
257 {
258  using Elem = typename MpmcQueue<ValueT>::Elem;
259  using Impl = typename MpmcQueue<ValueT>::Impl;
260 
261  assert(capacity >= 2);
262 
263  capacity = next_pow2(capacity);
264  const auto size = sizeof(Impl) + capacity * sizeof(Elem);
265 
266  os::ftruncate(fh.get(), size);
267  Mmap mem_map{os::mmap(nullptr, size, PROT_READ | PROT_WRITE, MAP_SHARED, fh.get(), 0)};
268  auto* const impl = static_cast<Impl*>(mem_map.get().data());
269 
270  std::memset(impl, 0, size);
271  // Initialise sequence numbers.
272  for (std::int64_t i{0}; i < static_cast<std::int64_t>(capacity); ++i) {
273  __atomic_store_n(&impl->elems[i].seq, i, __ATOMIC_RELAXED);
274  }
275 }
