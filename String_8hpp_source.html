182 {
183  const std::size_t len{std::min(n, src.size())};
184  if (len > 0) {
185  std::memcpy(dst, src.data(), len);

187  if (len < n) {

199 template <char PadC>
200 inline std::size_t pstrcpyid(char* dst, std::int64_t id, std::size_t n) noexcept
201 {
202  const auto end = dst + n;
203  const auto [ptr, ec] = std::to_chars(dst, end, id);
204  if (ec == std::errc::value_too_large) {
205  return 0;
206  }
207  if (ptr < end) {
208  std::memset(ptr, PadC, end - ptr);
209  }
210  return ptr - dst;
211 }
