139template <char PadC>
140constexpr std::size_t pstrlen(const char* src, std::size_t n) noexcept
141{
142 if constexpr (PadC == '\0') {
143 // Optimised case.
144 return strnlen(src, n);
145 } else {
146 std::size_t i{0};
147 while (i < n && src[i] != PadC) {
148 ++i;
149 }
150 return i;
151 }
152}

160template <char PadC>
161constexpr std::size_t pstrcpy(char* dst, const char* src, std::size_t n) noexcept
162{
163 if constexpr (PadC == '\0') {
164#ifndef __clang__
165#pragma GCC diagnostic push
166#pragma GCC diagnostic ignored "-Wstringop-truncation"
167#endif
168 // Optimised case.
169 return stpncpy(dst, src, n) - dst;
170#ifndef __clang__
171#pragma GCC diagnostic pop
172#endif
173 } else {
174 std::size_t i{0};
175 for (; i < n && src[i] != '\0'; ++i) {
176 dst[i] = src[i];
177 }
178 if (i < n) {
179 std::memset(dst + i, PadC, n - i);
180 }
181 return i;
182 }
183}

191template <char PadC>
192constexpr std::size_t pstrcpy(char* dst, std::string_view src, std::size_t n) noexcept
193{
194 const std::size_t len{std::min(n, src.size())};
195 if (len > 0) {
196 std::memcpy(dst, src.data(), len);
197 }
198 if (len < n) {
199 std::memset(dst + len, PadC, n - len);
200 }
201 return len;
202}

210template <char PadC>
211constexpr std::size_t pstrcpyid(char* dst, std::int64_t id, std::size_t n) noexcept
212{
213 auto* end = dst + n;
214 const auto [eptr, ec] = std::to_chars(dst, end, id);
215 if (ec == std::errc::value_too_large) {
216 return 0;
217 }
218 if (eptr < end) {
219 std::memset(eptr, PadC, end - eptr);
220 }
221 return eptr - dst;
222}
