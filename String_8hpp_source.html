144template <char PadC>
145constexpr std::size_t pstrlen(const char* src, std::size_t n) noexcept
146{
147 if constexpr (PadC == '\0') {
148 // Optimised case.
149 return strnlen(src, n);
150 } else {
151 std::size_t i{0};
152 while (i < n && src[i] != PadC) {
153 ++i;
154 }
155 return i;
156 }

176template <char PadC>
177constexpr std::size_t pstrcpy(char* dst, const char* src, std::size_t n) noexcept
178{
179 if constexpr (PadC == '\0') {
180#ifndef __clang__
181#pragma GCC diagnostic push
182#pragma GCC diagnostic ignored "-Wstringop-truncation"
183#endif
184 // Optimised case.
185 return stpncpy(dst, src, n) - dst;
186#ifndef __clang__
187#pragma GCC diagnostic pop
188#endif
189 } else {
190 std::size_t i{0};
191 for (; i < n && src[i] != '\0'; ++i) {
192 dst[i] = src[i];
193 }
194 if (i < n) {
195 std::memset(dst + i, PadC, n - i);
196 }
197 return i;
198 }
199}

244template <char PadC>
245constexpr std::size_t pstrcpyid(char* dst, std::int64_t id, std::size_t n) noexcept
246{
247 auto* end = dst + n;
248 const auto [eptr, ec] = std::to_chars(dst, end, id);
249 if (ec == std::errc::value_too_large) {
250 return 0;
251 }
252 if (eptr < end) {
253 std::memset(eptr, PadC, end - eptr);
254 }
255 return eptr - dst;
256}
