43  /*
44  * Returns true if connection was established synchronously or false if connection is pending
45  * asynchronous completion.
46  */
47  bool connect(CyclTime now, Reactor& r, const Endpoint& ep)
48  {
49  StreamSockClnt sock{ep.protocol()};
50  sock.set_non_block();
51  if (sock.is_ip_family()) {
52  set_tcp_no_delay(sock.get(), true);
53  }
54 
55  std::error_code ec;
56  sock.connect(ep, ec);
57  if (ec) {
58  if (ec.value() != EINPROGRESS) {
59  throw std::system_error{ec, "connect"};
60  }
61  sub_
62  = r.subscribe(*sock, EventIn | EventOut, bind<&StreamConnector::on_io_event>(this));
63  ep_ = ep;
64  sock_ = std::move(sock);
65  return false;
66  }
67  static_cast<DerivedT*>(this)->do_connect(now, std::move(sock), ep);
68  return true;
69  }

71  protected:
72  ~StreamConnector() = default;
73 
74  private:
75  void on_io_event(CyclTime now, int fd, unsigned events)
76  {
77  IoSock sock{std::move(sock_)};
78  sub_.reset();
79  try {
80  const auto ec = sock.get_so_error();
81  if (ec) {
82  throw std::system_error{ec, "connect"};
83  }
84  static_cast<DerivedT*>(this)->do_connect(now, std::move(sock), ep_);
85  } catch (const std::exception& e) {
86  static_cast<DerivedT*>(this)->do_connect_error(now, e);
87  }
88  }
